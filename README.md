### `POST /api/split-pdf` — Main Split Endpoint

**Request (multipart/form-data):**

| Parameter | Type | Required | Default | Description |
|-----------|------|----------|---------|-------------|
| `file` | File | ✅ | — | PDF file — any size, any version |

## 🐍 Backend Engine

### Lossless Quality Cascade

```python
# pdf_split.py — _split_lossless_single_page()
# ⚠️ NEVER uses Ghostscript — it always re-encodes images

def _split_lossless_single_page(src_path, out_path, page_idx, password=None):
    # ── Attempt 1: pikepdf — byte-perfect, zero re-encoding ──────────
    try:
        with pikepdf.open(src_path, password=password or '') as pdf:
            new_pdf = pikepdf.new()
            new_pdf.pages.append(pdf.pages[page_idx])
            new_pdf.save(out_path, recompress_flate=False)  # ← KEY: no re-encoding
        return 'pikepdf'
    except Exception:
        pass

    # ── Attempt 2: PyMuPDF — fast, excellent fidelity ────────────────
    try:
        src = fitz.open(src_path)
        if password:
            src.authenticate(password)
        out = fitz.open()
        out.insert_pdf(src, from_page=page_idx, to_page=page_idx)
        out.save(out_path, garbage=0, deflate=False)  # ← no recompression
        return 'fitz'
    except Exception:
        pass

    # ── Attempt 3: pypdf — universal fallback ────────────────────────
    reader = PdfReader(src_path)
    if password:
        reader.decrypt(password)
    writer = PdfWriter()
    writer.add_page(reader.pages[page_idx])
    with open(out_path, 'wb') as f:
        writer.write(f)
    return 'pypdf'
```

### Parallel Processing

```python
from concurrent.futures import ThreadPoolExecutor, as_completed

# All Pages mode — parallel burst:
def _split_all_parallel(src_path, out_dir, page_count, naming, password, job_id):
    paths = [os.path.join(out_dir, f"{naming.format(n=i+1)}.pdf")
             for i in range(page_count)]

    with ThreadPoolExecutor(max_workers=min(8, page_count)) as ex:
        futures = {
            ex.submit(_split_lossless_single_page, src_path, paths[i], i, password): i
            for i in range(page_count)
        }
        for future in as_completed(futures):
            idx = futures[future]
            engine = future.result()  # 'pikepdf' | 'fitz' | 'pypdf'
            if job_id:
                _progress_queues[job_id].put(('page_done', idx, engine))

    return paths
```

### Blank Page Detection

```python
# Uses PyMuPDF pixel analysis — no external deps, no Tesseract

def _is_blank_page(page, threshold=0.97):
    """Returns True if >threshold% of pixels are white (>245 brightness)."""
    pix = page.get_pixmap(dpi=24, colorspace=fitz.csGRAY)
    samples = list(pix.samples)
    white_ratio = sum(1 for b in samples if b > 245) / max(len(samples), 1)
    return white_ratio >= threshold
```
